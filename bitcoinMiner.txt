    *microbitcoinLog
        always hidden
        // log10 of current uBTC using lim formula with n=10^9
        is always:(max(0, (floor((pow(microbitcoins, pow(10, -9)) - 1) / (pow(10, pow(10, -9)) - 1)))))
    
    *dollarsLog
        always hidden
        // log10 of current $ using lim formula with n=10^9
        is always:(max(0, (floor((pow(dollars, pow(10, -9)) - 1) / (pow(10, pow(10, -9)) - 1)))))
    
    *bitcoinExchangingBTC
        always hidden
        is always:max(100, pow(10, microbitcoinLog))
    
    *bitcoinExchangingDol
        always hidden
        is always:max(2, (2 * pow(10, microbitcoinLog - 2)))
    
    *dollarsExchangingDol
        always hidden
        is always:(pow(10, dollarsLog))
    
    *dollarsExchangingBTC
        always hidden
        is always:(4.9 * pow(10, dollarsLog + 1))

Shinies
	*luckyClover
		movement:onRight moveLeft wiggle:0.25 growShrink
		frequency:90
		frequency variation:20
		icon:icons[3,15]
		class:bigButton
		on click:
			$handicaps=random(0,5)
			if (chance(70%))
				yield 61 luckyTiming
				toast You got lucky! Duplicate the productivity of your buildings for 1 minute.
			else
				if ($handicaps=0)
					yield 31 ineffTiming
					toast You have installed an inefficient program. This decreases the productivity of your CPUs by 10% for 30 seconds.
				else if ($handicaps=1)
					yield 61 overTiming
					toast Your CPU is overheating, so you need to power it off and wait. This decreases the productivity of your CPUs by 20% for 1 minute.
				else if ($handicaps=2)
					yield 61 malWTiming
					toast You have a Malware! Somebody is stealing your resources. This decreases your total productivity by 50% for 1 minute.
				else if ($handicaps=3)
					yield 31 frozenTiming
					toast Your system is frozen. You should reboot it. This decreases your total productivity by 50% for 30 seconds.
				else if ($handicaps=4)
					yield 11 cutTiming
					toast A power cut in your city causes you tremendous losses. This decreases your total productivity by 70% for 10 seconds.
				else 
					yield 121 electricTiming
					toast The electricity price has been raised this month, so your production is now less profitable. This decreases your total productivity by 10% for 2 minutes.
				end
			end
		end
